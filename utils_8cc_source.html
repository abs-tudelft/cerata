35 std::string ToString(const std::unordered_map<std::string, std::string> &meta) {
36  std::string result;
37  if (!meta.empty()) {
38  result += "{";
39  size_t i = 0;
40  for (const auto &kv : meta) {
41  result += kv.first + "=" + kv.second;
42  if (i != meta.size() - 1) {
43  result += ",";
44  }
45  i++;
46  }
47  result += "}";
48  }
49  return result;
50 }
