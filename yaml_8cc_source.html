37 auto YamlConverter::Visit(const YAML::Node &node) -> Status {
38  try {
39  switch (node.Type()) {
40  case YAML::NodeType::Sequence: {
41  for (const auto &item : node) {
42  if (item[YAML_KEY_FIELD] || item[YAML_KEY_RECORD]) {
43  std::shared_ptr<cerata::Field> field;
44  auto conv = YamlConverter(item, &field);
45  CERATA_ROE(conv.Convert());
46  fields_.push_back(field);
47  } else {
48  return Status(Error::YAMLError,
49  "Sequence can only contain \"" + std::string(YAML_KEY_FIELD) + "\" or \""
50  + std::string(YAML_KEY_RECORD) + "\"");
51  }
52  }
53  break;
54  }
55 
56  case YAML::NodeType::Map: {
57  if (node[YAML_KEY_NAME]) {
58  name_ = node[YAML_KEY_NAME].as<std::string>();
59  } else {
60  return Status(Error::YAMLError, "Field must have \"name\"");
61  }
62  if (node[YAML_KEY_WIDTH]) {
63  width_ = node[YAML_KEY_WIDTH].as<unsigned int>();
64  }
65  if (node[YAML_KEY_VECTOR]) {
66  force_vector_ = node[YAML_KEY_VECTOR].as<bool>();
67  }
68  if (node[YAML_KEY_REVERSE]) {
69  reverse_ = node[YAML_KEY_REVERSE].as<bool>();
70  }
71  if (node[YAML_KEY_FIELDS] && (width_ == 0)) {
72  auto fields = node[YAML_KEY_FIELDS];
73  CERATA_ROE(Visit(node[YAML_KEY_FIELDS]));
74  } else if (node[YAML_KEY_FIELDS] && (width_ > 0)) {
75  return Status(Error::YAMLError, "Field " + name_ + " cannot have both width and subfields.");
76  } else if (!node[YAML_KEY_WIDTH]) {
77  return Status(Error::YAMLError, "Field " + name_ + " must have either width or subfields.");
78  }
79  break;
80  }
81 
82  // Some nodes we should never encounter through this visitor.
83  case YAML::NodeType::Undefined: {
84  return Status(Error::YAMLError,
85  "Unexpected undefined at " + std::to_string(node.Mark().line) + ":"
86  + std::to_string(node.Mark().column));
87  }
88  case YAML::NodeType::Null: {
89  return Status(Error::YAMLError, "Unexpected null at " + ToString(node.Mark()));
90  }
91  case YAML::NodeType::Scalar: {
92  return Status(Error::YAMLError, "Unexpected scalar at " + ToString(node.Mark()));
93  }
94  }
95  } catch (const YAML::Exception &e) {
96  return Status(Error::YAMLError, "YAML parsing error: " + std::string(e.msg));
97  }
98  return Status::OK();
99 }

101 auto YamlConverter::Convert() -> Status {
102  auto status = Visit(root_);
103  if (!status.ok()) { return status; }
104 
105  if (fields_.empty()) {
106  // Check if this is to be a record, vector or just a bit.
107  if ((width_ > 1) || force_vector_) {
108  *out_ = cerata::field(name_, cerata::vector(width_), reverse_);
109  } else if (width_ > 0) {
110  *out_ = cerata::field(name_, cerata::bit(name_), reverse_);
111  } else {
112  return Status(Error::YAMLError, "Width for " + name_ + " must be greater than 0.");
113  }
114  } else {
115  *out_ = cerata::field(name_, cerata::record(name_, {fields_}), reverse_);
116  }
117  return Status::OK();
118 }
